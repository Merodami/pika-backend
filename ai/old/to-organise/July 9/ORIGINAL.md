     const cacheKey = `availability:${gymId}:${date.toISOString()}`

     // L1 Cache check

     if (slots) return slots

     // L2 Cache check
     slots = await this.redis.get(cacheKey)
     if (slots) {

     this.memoryCache.set(cacheKey, slots, 300)

     return slots
   }
